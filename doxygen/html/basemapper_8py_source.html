94 class BytesIOBoundaryHandler(BoundaryHandler):
95  """Extracts BBOX from GeoJSON data stored in a BytesIO object."""
96 
97  def __init__(self, boundary: BytesIO):
98  """Initialize the BytesIOBoundaryHandler with a BytesIO input."""
99  self.boundaryboundary = boundary
100 
101  def make_bbox(self) -> BoundingBox:
102  """Extract and return the bounding box from the GeoJSON data.
103 
104  Returns:
105  BoundingBox: The bounding box as a tuple (min_x, min_y, max_x, max_y).
106  """
107  log.debug(f"Reading geojson BytesIO : {self.boundary}")
108  # Rewind the BytesIO object to the beginning before passing it to geojson.load()
109  self.boundaryboundary.seek(0)
110  with self.boundaryboundary as buffer:
111  poly = geojson.load(buffer)
112 
113  if "features" in poly:
114  geometry = shape(poly["features"][0]["geometry"])
115  elif "geometry" in poly:
116  geometry = shape(poly["geometry"])
117  else:
118  geometry = shape(poly)
119 
120  if isinstance(geometry, list):
121  # Multiple geometries
122  log.debug("Creating union of multiple bbox geoms")
123  geometry = unary_union(geometry)
124 
125  if geometry.is_empty:
126  msg = f"No bbox extracted from {geometry}"
127  log.error(msg)
128  raise ValueError(msg) from None
129 
130  bbox = geometry.bounds
131  # left, bottom, right, top
132  # minX, minY, maxX, maxY
133  return bbox

165 def format_url(site: dict, tile: tuple) -> Optional[str]:
166  """Format the URL for the given site and tile.
167 
168  Args:
169  site (dict): The site configuration with URL and source type.
170  tile (tuple): The tile coordinates (x, y, z).
171 
172  Returns:
173  str: The formatted URL or None if the source is unsupported.
174  """
175  source_url = site["url"]
176  if site.get("xy"):
177  # z/x/y format download, move to z/y/x structure on disk
178  url_path = f"{tile[2]}/{tile[0]}/{tile[1]}"
179  else:
180  # z/y/x format download, keep the same on disk
181  url_path = f"{tile[2]}/{tile[1]}/{tile[0]}"
182 
183  match site["source"]:
184  # NOTE we use % syntax to replace the placeholder %s
185  case "esri":
186  return source_url % url_path
187  case "bing":
188  bingkey = mercantile.quadkey(tile)
189  return source_url % bingkey
190  case "topo":
191  # FIXME does this work an intended?
192  return source_url % f"{tile[2]}/{tile[1]}/{tile[0]}"
193  case "google":
194  return source_url % f"x={tile[0]}&s=&y={tile[1]}&z={tile[2]}"
195  case "oam":
196  return source_url % url_path
197  case "custom":
198  return source_url % url_path
199  case _:
200  log.error(f"Unsupported source: {site['source']}")
201  return None

204 def download_tile(dest: str, tile: tuple, mirrors: list[dict]) -> None:
205  """Download a single tile from the given list of mirrors.
206 
207  Args:
208  dest (str): The destination directory.
209  tile (tuple): The tile coordinates (x, y, z).
210  mirrors (list): The list of mirrors to get imagery.
211  """
212  for site in mirrors:
213  download_url = format_url(site, tile)
214  if download_url:
215  filespec = f"{tile[2]}/{tile[1]}/{tile[0]}.{site['suffix']}"
216  outfile = Path(dest) / filespec
217  if not outfile.exists():
218  try:
219  log.debug(f"Attempting URL download: {download_url}")
220  dl = SmartDL(download_url, dest=str(outfile), connect_default_logger=False)
221  dl.start()
222  return
223  except Exception as e:
224  log.error(e)
225  log.error(f"Couldn't download file for {filespec}")
226  else:
227  log.debug(f"{outfile} exists!")
228  else:
229  continue

233  """Thread to handle downloads for Queue.
234 
235  Args:
236  dest (str): The filespec of the tile cache.
237  mirrors (list): The list of mirrors to get imagery.
238  tiles (list): The list of tiles to download.
239  """
240  if len(tiles) == 0:
241  # epdb.st()
242  return
243 
244  # Create the subdirectories as pySmartDL doesn't do it for us
245  Path(dest).mkdir(parents=True, exist_ok=True)
246

295  def customTMS(self, url: str, is_oam: bool = False, is_xy: bool = False):
296  """Add a custom TMS URL to the list of sources.
297 
298  The url must end in %s to be replaced with the tile xyz values.
299 
300  Format examples:
301  https://basemap.nationalmap.gov/ArcGIS/rest/services/USGSTopo/MapServer/tile/{z}/{y}/{x}
302  https://maps.nyc.gov/xyz/1.0.0/carto/basemap/%s
303  https://maps.nyc.gov/xyz/1.0.0/carto/basemap/{z}/{x}/{y}.jpg
304 
305  The method will replace {z}/{x}/{y}.jpg with %s
306 
307  Args:
308  url (str): The URL string
309  source (str): The provier source, for setting attribution
310  is_xy (bool): Swap the x and y for the provider --> 'zxy'
311  """
312  # Remove any file extensions if present and update the 'suffix' parameter
313  # NOTE the file extension gets added again later for the download URL
314  if url.endswith(".jpg"):
315  suffix = "jpg"
316  url = url[:-4] # Remove the last 4 characters (".jpg")
317  elif url.endswith(".png"):
318  suffix = "png"
319  url = url[:-4] # Remove the last 4 characters (".png")
320  else:
321  # FIXME handle other formats for custom TMS
322  suffix = "jpg"
323 
324  # If placeholders present, validate they have no additional spaces
325  if "{" in url and "}" in url:
326  pattern = r".*/\{[zxy]\}/\{[zxy]\}/\{[zxy]\}(?:/|/?)"
327  if not bool(re.search(pattern, url)):
328  msg = "Invalid TMS URL format. Please check the URL placeholders {z}/{x}/{y}."
329  log.error(msg)
330  raise ValueError(msg)
331 
332  # Remove "{z}/{x}/{y}" placeholders if they are present
333  url = re.sub(r"/{[xyz]+\}", "", url)
334  # Append "%s" to the end of the URL to later add the tile path
335  url = url + r"/%s"
336 
337  if is_oam:
338  # Override dummy OAM URL
339  source = "oam"
340  self.sourcessources[source]["url"] = url
341  else:
342  source = "custom"
343  tms_params = {"name": source, "url": url, "suffix": suffix, "source": source, "xy": is_xy}
344  log.debug(f"Setting custom TMS with params: {tms_params}")
345  self.sourcessources[source] = tms_params
346 
347  # Select the source
348  self.sourcesource = source

358  def getTiles(self, zoom: int) -> int:
359  """Get a list of tiles for the specified zoom level.
360 
361  Args:
362  zoom (int): The Zoom level of the desired map tiles.
363 
364  Returns:
365  int: The total number of map tiles downloaded.
366  """
367  info = get_cpu_info()
368  cores = info["count"]
369 
370  self.tilestiles = list(mercantile.tiles(self.bboxbbox[0], self.bboxbbox[1], self.bboxbbox[2], self.bboxbbox[3], zoom))
371  total = len(self.tilestiles)
372  log.info(f"{total} tiles for zoom level {zoom}")
373 
374  mirrors = [self.sourcessources[self.sourcesource]]
375  chunk_size = max(1, round(total / cores))
376 
377  if total < chunk_size or chunk_size == 0:
378  dlthread(self.basebase, mirrors, self.tilestiles)
379  else:
380  with concurrent.futures.ThreadPoolExecutor(max_workers=cores) as executor:
381  futures = []
382  for i in range(0, total, chunk_size):
383  chunk = self.tilestiles[i : i + chunk_size]
384  futures.append(executor.submit(dlthread, self.basebase, mirrors, chunk))
385  log.debug(f"Dispatching Block {i}:{i + chunk_size}")
386  concurrent.futures.wait(futures)
387 
388  return total

411 def tileid_from_zyx_dir_path(filepath: Union[Path, str]) -> int:
412  """Helper function to get the tile id from a tile in xyz (zyx) directory structure.
413 
414  TMS typically has structure z/y/x.png
415  If the --xy flag was used previously, the TMS was downloaed into
416  directories of z/y/x structure from their z/x/y URL.
417 
418  Args:
419  filepath (Union[Path, str]): The path to tile image within the xyz directory.
420 
421  Returns:
422  int: The globally defined tile id from the xyz definition.
423  """
424  # Extract the final 3 parts from the TMS file path
425  tile_image_path = Path(filepath).parts[-3:]
426 
427  try:
428  final_tile = int(Path(tile_image_path[-1]).stem)
429  except ValueError as e:
430  msg = f"Invalid tile path (cannot parse as int): {str(tile_image_path)}"
431  log.error(msg)
432  raise ValueError(msg) from e
433 
434  x = final_tile
435  z, y = map(int, tile_image_path[:-1])
436 
437  return zxy_to_tileid(z, x, y)

440 def tile_dir_to_pmtiles(
441  outfile: str,
442  tile_dir: str | Path,
443  bbox: tuple,
444  image_format: str,
445  zoom_levels: list[int],
446  attribution: str,
447 ):
448  """Write PMTiles archive from tiles in the specified directory.
449 
450  Args:
451  outfile (str): The output PMTiles archive file path.
452  tile_dir (str | Path): The directory containing the tile images.
453  bbox (tuple): Bounding box in format (min_lon, min_lat, max_lon, max_lat).
454  attribution (str): Attribution string to include in PMTile archive.
455 
456  Returns:
457  None
458  """
459  tile_dir = Path(tile_dir)
460 
461  # Abort if no files are present
462  first_file = next((file for file in tile_dir.rglob("*.*") if file.is_file()), None)
463  if not first_file:
464  err = "No tile files found in the specified directory. Aborting PMTile creation."
465  log.error(err)
466  raise ValueError(err)
467 
468  tile_format = image_format.upper()
469  # NOTE JPEG exception / flexible extension (.jpg, .jpeg)
470  if tile_format == "JPG":
471  tile_format = "JPEG"
472  log.debug(f"PMTile determind internal file format: {tile_format}")
473  possible_tile_formats = [f".{e.name.lower()}" for e in PMTileType]
474  possible_tile_formats.append(".jpg")
475  possible_tile_formats.remove(".unknown")
476 
477  with open(outfile, "wb") as pmtile_file:
478  writer = PMTileWriter(pmtile_file)
479 
480  for tile_path in tile_dir.rglob("*"):
481  if tile_path.is_file() and tile_path.suffix.lower() in possible_tile_formats:
482  tile_id = tileid_from_zyx_dir_path(tile_path)
483 
484  with open(tile_path, "rb") as tile:
485  writer.write_tile(tile_id, tile.read())
486 
487  min_lon, min_lat, max_lon, max_lat = bbox
488  log.debug(
489  f"Writing PMTiles file with min_zoom ({zoom_levels[0]}) "
490  f"max_zoom ({zoom_levels[-1]}) bbox ({bbox}) tile_compression None"
491  )
492 
493  # Write PMTile metadata
494  writer.finalize(
495  header={
496  "tile_type": PMTileType[tile_format],
497  "tile_compression": PMTileCompression.NONE,
498  "min_zoom": zoom_levels[0],
499  "max_zoom": zoom_levels[-1],
500  "min_lon_e7": int(min_lon * 10000000),
501  "min_lat_e7": int(min_lat * 10000000),
502  "max_lon_e7": int(max_lon * 10000000),
503  "max_lat_e7": int(max_lat * 10000000),
504  "center_zoom": zoom_levels[0],
505  "center_lon_e7": int(min_lon + ((max_lon - min_lon) / 2)),
506  "center_lat_e7": int(min_lat + ((max_lat - min_lat) / 2)),
507  },
508  metadata={"attribution": f"© {attribution}"},
509  )

512 def create_basemap_file(
513  boundary=None,
514  tms=None,
515  xy=False,
516  outfile=None,
517  zooms="12-17",
518  outdir=None,
519  source="esri",
520  append: bool = False,
521 ) -> None:
522  """Create a basemap with given parameters.
523 
524  Args:
525  boundary (str | BytesIO, optional): The boundary for the area you want.
526  tms (str, optional): Custom TMS URL.
527  xy (bool, optional): Swap the X & Y coordinates when using a
528  custom TMS if True.
529  outfile (str, optional): Output file name for the basemap.
530  zooms (str, optional): The Zoom levels, specified as a range
531  (e.g., "12-17") or comma-separated levels (e.g., "12,13,14").
532  outdir (str, optional): Output directory name for tile cache.
533  source (str, optional): Imagery source, one of
534  ["esri", "bing", "topo", "google", "oam", "custom"] (default is "esri").
535  append (bool, optional): Whether to append to an existing file
536 
537  Returns:
538  None
539  """
540  log.debug(
541  "Creating basemap with params: "
542  f"boundary={boundary} | "
543  f"outfile={outfile} | "
544  f"zooms={zooms} | "
545  f"outdir={outdir} | "
546  f"source={source} | "
547  f"tms={tms}"
548  )
549 
550  # Validation
551  if not boundary:
552  err = "You need to specify a boundary! (in-memory object or bbox)"
553  log.error(err)
554  raise ValueError(err)
555 
556  # Get all the zoom levels we want
557  zoom_levels = list()
558  if zooms:
559  if zooms.find("-") > 0:
560  start = int(zooms.split("-")[0])
561  end = int(zooms.split("-")[1]) + 1
562  x = range(start, end)
563  for i in x:
564  zoom_levels.append(i)
565  elif zooms.find(",") > 0:
566  levels = zooms.split(",")
567  for level in levels:
568  zoom_levels.append(int(level))
569  else:
570  zoom_levels.append(int(zooms))
571 
572  if not outdir:
573  base = Path.cwd().absolute()
574  else:
575  base = Path(outdir).absolute()
576 
577  # Source / TMS validation
578  if not source and not tms:
579  err = "You need to specify a source!"
580  log.error(err)
581  raise ValueError(err)
582  if source == "oam" and not tms:
583  err = "A TMS URL must be provided for OpenAerialMap!"
584  log.error(err)
585  raise ValueError(err)
586  # A custom TMS provider
587  if source != "oam" and tms:
588  source = "custom"
589 
590  tiledir = base / f"{source}tiles"
591  # Make tile download directory
592  tiledir.mkdir(parents=True, exist_ok=True)
593  # Convert to string for other methods
594  tiledir = str(tiledir)
595 
596  basemap = BaseMapper(boundary, tiledir, source)
597 
598  if tms:
599  # Add TMS URL to sources for download
600  basemap.customTMS(tms, True if source == "oam" else False, xy)
601 
602  # Args parsed, main code:
603  tiles = list()
604  for zoom_level in zoom_levels:
605  # Download the tile directory
606  basemap.getTiles(zoom_level)
607  tiles += basemap.tiles
608 
609  if not outfile:
610  log.info(f"No outfile specified, tile download finished: {tiledir}")
611  return
612 
613  suffix = Path(outfile).suffix.lower()
614  image_format = basemap.sources[source].get("suffix", "jpg")
615  log.debug(f"Basemap output format: {suffix} | Image format: {image_format}")
616 
617  if any(substring in suffix for substring in ["sqlite", "mbtiles"]):
618  outf = DataFile(outfile, basemap.getFormat(), append)
619  if suffix == ".mbtiles":
620  outf.addBounds(basemap.bbox)
621  outf.addZoomLevels(zoom_levels)
622  # Create output database and specify image format, png, jpg, or tif
623  outf.writeTiles(tiles, tiledir, image_format)
624 
625  elif suffix == ".pmtiles":
626  tile_dir_to_pmtiles(outfile, tiledir, basemap.bbox, image_format, zoom_levels, source)
627 
628  else:
629  msg = f"Format {suffix} not supported"
630  log.error(msg)
631  raise ValueError(msg) from None
632  log.info(f"Wrote {outfile}")

635 def move_tiles(
636  boundary=None,
637  indir=None,
638  outdir=None,
639 ) -> None:
640  """Move tiles within a boundary to another directory. Used for managing the
641  map tile cache.
642 
643  Args:
644  boundary (str | BytesIO, optional): The boundary for the area you want.
645  indir (str, optional): Top level directory for existing tile cache.
646  outdir (str, optional): Output directory name for the new tile cache.
647 
648  Returns:

653  zooms = os.listdir(indir)

656  log.info(f"Making {outdir}...")
