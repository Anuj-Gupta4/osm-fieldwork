81  def standardize_language_columns(df):
82  """Standardize existing language columns.
83 
84  :param df: Original DataFrame with existing translations.
85  :param DEFAULT_LANGAUGES: List of DEFAULT_LANGAUGES with their short codes, e.g., {"english": "en", "french": "fr"}.
86  :param base_columns: List of base columns to check (e.g., 'label', 'hint', 'required_message').
87  :return: Updated DataFrame with standardized and complete language columns.
88  """
89  base_columns = ["label", "hint", "required_message"]
90  df.columns = df.columns.str.lower()
91  existing_columns = df.columns.tolist()
92 
93  # Map existing columns and standardize their names
94  for col in existing_columns:
95  standardized_col = col
96  for base_col in base_columns:
97  if col.startswith(f"{base_col}::"):
98  match = re.match(rf"{base_col}::(\w+)", col)
99  if match:
100  lang_name = match.group(1)
101  if lang_name in DEFAULT_LANGUAGES:
102  standardized_col = f"{base_col}::{lang_name}({DEFAULT_LANGUAGES[lang_name]})"
103 
104  elif col == base_col: # if only label,hint or required_message then add '::english(en)'
105  standardized_col = f"{base_col}::english(en)"
106 
107  if col != standardized_col:
108  df.rename(columns={col: standardized_col}, inplace=True)
109  return df

111  def filter_df_empty_rows(df: pd.DataFrame, column: str = NAME_COLUMN):
112  """Remove rows with None values in the specified column.
113 
114  NOTE We retain 'end group' and 'end group' rows even if they have no name.
115  NOTE A generic df.dropna(how="all") would not catch accidental spaces etc.
116  """
117  if column in df.columns:
118  # Only retain 'begin group' and 'end group' if 'type' column exists
119  if "type" in df.columns:
120  return df[(df[column].notna()) | (df["type"].isin(["begin group", "end group", "begin_group", "end_group"]))]
121  else:
122  return df[df[column].notna()]

166 def merge_dataframes(mandatory_df: pd.DataFrame, user_question_df: pd.DataFrame, digitisation_df: pd.DataFrame) -> pd.DataFrame:
167  """Merge multiple Pandas dataframes together, removing duplicate fields."""
168  if "list_name" in user_question_df.columns:
169  merged_df = pd.concat(
170  [
171  mandatory_df,
172  user_question_df,
173  digitisation_df,
174  ],
175  ignore_index=True,
176  )
177  # NOTE here we remove duplicate PAIRS based on `list_name` and the name column
178  # we have `allow_duplicate_choices` set in the settings sheet, so it is possible
179  # to have duplicate NAME_COLUMN entries, if they are in different a `list_name`.
180  return merged_df.drop_duplicates(subset=["list_name", NAME_COLUMN], ignore_index=True)
181 
182  user_question_df = user_question_df.apply(normalize_with_meta, axis=1, meta_df=meta_df)
183 
184  # Find common fields between user_question_df and mandatory_df or digitisation_df
185  duplicate_fields = set(user_question_df[NAME_COLUMN]).intersection(
186  set(mandatory_df[NAME_COLUMN]).union(set(digitisation_df[NAME_COLUMN]))
187  )
188 
189  # NOTE filter out 'end group' from duplicate check as they have empty NAME_COLUMN
190  end_group_rows = user_question_df[user_question_df["type"].isin(["end group", "end_group"])]
191  user_question_df_filtered = user_question_df[
192  (~user_question_df[NAME_COLUMN].isin(duplicate_fields)) | (user_question_df.index.isin(end_group_rows.index))
193  ]
194 
195  # Create survey group wrapper
196  survey_group = create_survey_group(SURVEY_GROUP_NAME)
197 
198  # Concatenate dataframes in the desired order
199  return pd.concat(
200  [
201  mandatory_df,
202  # Wrap the survey question in a group
203  survey_group["begin"],
204  user_question_df_filtered,
205  survey_group["end"],
206  digitisation_df,
207  ],
208  ignore_index=True,

274 
275  if "survey" not in custom_sheets:
276  msg = "Survey sheet is required in XLSForm!"
277  log.error(msg)
278  raise ValueError(msg)
279 
280  custom_sheets = standardize_xlsform_sheets(custom_sheets)
281 
282  log.debug("Merging survey sheet XLSForm data")
283  custom_sheets["survey"] = merge_dataframes(survey_df, custom_sheets.get("survey"), digitisation_df)
284  # Hardcode the form_category value for the start instructions
285  if form_category.endswith("s"):
286  # Plural to singular
287  form_category = form_category[:-1]
288  form_category_row = custom_sheets["survey"].loc[custom_sheets["survey"]["name"] == "form_category"]
289  if not form_category_row.empty:
290  custom_sheets["survey"].loc[custom_sheets["survey"]["name"] == "form_category", "calculation"] = f"once('{form_category}')"
291 
292  # Ensure the 'choices' sheet exists in custom_sheets
293  if "choices" not in custom_sheets or custom_sheets["choices"] is None:
294  custom_sheets["choices"] = pd.DataFrame(columns=["list_name", "name", "label::english(en)"])
295 
296  log.debug("Merging choices sheet XLSForm data")
297  custom_sheets["choices"] = merge_dataframes(choices_df, custom_sheets.get("choices"), digitisation_choices_df)
298 
299  # Append or overwrite 'entities' and 'settings' sheets
300  log.debug("Overwriting entities and settings XLSForm sheets")
301  custom_sheets["entities"] = entities_df
302  custom_sheets["settings"] = settings_df
303  if "entities" not in custom_sheets:
304  msg = "Entities sheet is required in XLSForm!"
305  log.error(msg)
306  raise ValueError(msg)
307  if "settings" not in custom_sheets:
308  msg = "Settings sheet is required in XLSForm!"
309  log.error(msg)
310  raise ValueError(msg)
311 
312  # Set the 'version' column to the current timestamp (if 'version' column exists in 'settings')
313  xform_id = existing_id if existing_id else uuid4()
314  current_datetime = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
315  log.debug(f"Setting xFormId = {xform_id} | form title = {form_category} | version = {current_datetime}")
316  custom_sheets["settings"]["version"] = current_datetime
317  custom_sheets["settings"]["form_id"] = xform_id
318  custom_sheets["settings"]["form_title"] = form_category
319 
320  # Append select_one_from_file for additional entities
321  if additional_entities:
322  log.debug("Adding additional entity list reference to XLSForm")
323  for entity_name in additional_entities:
324  custom_sheets["survey"] = append_select_one_from_file_row(custom_sheets["survey"], entity_name)
325 
326  # Return spreadsheet wrapped as BytesIO memory object
327  output = BytesIO()
328  with pd.ExcelWriter(output, engine="openpyxl") as writer:
329  for sheet_name, df in custom_sheets.items():
330  df.to_excel(writer, sheet_name=sheet_name, index=False)
